```while read ip;do response=$(whois $ip | grep OrgName) && printf "[+]ip address: $ip \n[+]$response \n\n"; done < iplist.txt```

```while read site;do status=$(curl -I $site 2>/dev/null | head -1) && echo $site $status; done < iplist.txt```
